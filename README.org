*** HTTPS

    #+BEGIN_SRC sh
      mkdir -p ~/smart_cage_ws/src
      cd ~/smart_cage_ws
      wget https://raw.githubusercontent.com/janelia-ros/smart_cage_ros/master/smart_cage_https.repos
      vcs import src < smart_cage_https.repos
    #+END_SRC
